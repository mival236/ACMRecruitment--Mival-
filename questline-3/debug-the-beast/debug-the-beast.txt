def second_largest(arr):
    if len(arr) < 2:
        return None  # Not enough elements
    
    largest = second = float('-inf')
    for num in arr:
        if num > largest:
            second = largest
            largest = num
        elif largest > num > second:  # ensure num is strictly less than largest
            second = num
    
    return second if second != float('-inf') else None

if result is None:
    print("No valid second largest number.")
else:
    print("Second largest:", result)
